<script>

/* =========================================================

   GOOGLE DRIVE INTEGRATION

   ========================================================= */

const DRIVE_SCRIPT_URL = "https://script.google.com/macros/s/AKfycbwOfi0tL_kxeY1Cck0pvyy4-uySybqjLZL9cFpaTtO9KFnwGSUKibuj2TU0vegNjJxW/exec";


// Segundos que cada foto fica na tela

const SECONDS_PER_PHOTO = 8;


// Embaralhar ordem a cada carregamento? (true/false)

const SHUFFLE = true;


/* ========================================================= */


const stage = document.getElementById('stage');

const clockEl = document.getElementById('clock');

const dateEl = document.getElementById('date');

const statusEl = document.getElementById('status');

const errorBox = document.getElementById('errorBox');


let PHOTOS = [];

let order = [];

let idx = 0;

let currentEl = null;

let showNextTimeout = null;

let failedPhotos = new Set();


function preload(url) {

  return new Promise((resolve, reject) => {

    const img = new Image();

    img.onload = () => resolve(url);

    img.onerror = () => reject(url);

    img.src = url;

  });

}


function setStatus(message, duration = 3000) {

  statusEl.textContent = message;

  statusEl.classList.remove('fade');

  

  if (duration > 0) {

    setTimeout(() => {

      statusEl.classList.add('fade');

    }, duration);

  }

}


async function loadPhotosFromGoogleDrive() {

  try {

    setStatus("Carregando fotos do Drive...", 0);

    const response = await fetch(DRIVE_SCRIPT_URL);

    PHOTOS = await response.json();

    

    if (!Array.isArray(PHOTOS) || PHOTOS.length === 0) {

      throw new Error("Nenhuma foto encontrada");

    }

    

    setStatus(`✓ Carregadas ${PHOTOS.length} fotos`, 2000);

    initializeSlideshow();

  } catch (err) {

    setStatus("✗ Erro ao carregar fotos", 0);

    errorBox.style.display = 'flex';

    console.error("Drive error:", err);

  }

}


function initializeSlideshow() {

  order = [...PHOTOS.keys()];

  if (SHUFFLE) {

    for (let i = order.length - 1; i > 0; i--) {

      const j = Math.floor(Math.random() * (i + 1));

      [order[i], order[j]] = [order[j], order[i]];

    }

  }

  showNext();

}


async function showNext() {

  if (PHOTOS.length === 0) {

    errorBox.style.display = 'flex';

    return;

  }


  // Se todas as fotos falharam, mostrar erro

  if (failedPhotos.size === PHOTOS.length) {

    setStatus("✗ Todas as imagens falharam", 0);

    errorBox.style.display = 'flex';

    return;

  }


  const url = PHOTOS[order[idx % order.length]];

  idx++;


  try {

    await preload(url);

    failedPhotos.delete(url);

  } catch (e) {

    failedPhotos.add(url);

    setStatus(`Falha ao carregar (${failedPhotos.size}/${PHOTOS.length}), pulando...`);

    showNextTimeout = setTimeout(showNext, 300);

    return;

  }


  const el = document.createElement('div');

  el.className = 'slide active';

  el.style.backgroundImage = `url("${url}")`;

  stage.appendChild(el);


  if (currentEl) {

    const old = currentEl;

    old.classList.remove('active');

    old.classList.add('prev');

    setTimeout(() => old.remove(), 1500);

  }

  currentEl = el;


  showNextTimeout = setTimeout(showNext, SECONDS_PER_PHOTO * 1000);

}


// Relógio - atualiza a cada segundo

function tickClock() {

  const now = new Date();

  clockEl.textContent = now.toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' });

  dateEl.textContent = now.toLocaleDateString('pt-BR', {

    weekday: 'long', day: 'numeric', month: 'long'

  });

}

tickClock();

setInterval(tickClock, 1000);


// Evita a tela apagar/dormir enquanto a página estiver aberta

let wakeLock = null;

async function requestWakeLock() {

  try {

    if ('wakeLock' in navigator) {

      wakeLock = await navigator.wakeLock.request('screen');

      setStatus("Wake Lock ativado", 2000);

      wakeLock.addEventListener('release', () => {

        setStatus("Wake Lock liberado", 2000);

      });

    } else {

      setStatus("Wake Lock não suportado", 3000);

    }

  } catch (err) {

    setStatus("Wake Lock falhou", 3000);

  }

}

requestWakeLock();


// Reativa o wake lock se a aba voltar a ficar visível

document.addEventListener('visibilitychange', () => {

  if (document.visibilityState === 'visible') {

    requestWakeLock();

  }

});


// Limpar timeout ao descarregar página

window.addEventListener('beforeunload', () => {

  clearTimeout(showNextTimeout);

});


// Início - carregar fotos do Google Drive

loadPhotosFromGoogleDrive();
